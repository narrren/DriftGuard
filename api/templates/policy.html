# Define package for storage rules
package driftguard.storage

# Deny if S3 bucket is not encrypted
deny[msg] {
    input.resource.type == "aws_s3_bucket"
    not input.resource.values.server_side_encryption_configuration
    msg = sprintf("Bucket '%v' must be encrypted", [input.resource.name])
}

# Warn if public access is enabled
warn[msg] {
    input.resource.type == "aws_s3_bucket"
    input.resource.values.acl == "public-read"
    msg = "Public read access is discouraged"
}
